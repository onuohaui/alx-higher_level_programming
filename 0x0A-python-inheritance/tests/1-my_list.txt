#!/usr/bin/python3
MyList = __import__('1-my_list').MyList

def test_my_list():
    my_list = MyList()
    my_list.append(1)
    my_list.append(4)
    my_list.append(2)
    my_list.append(3)
    my_list.append(5)
    assert str(my_list) == '[1, 4, 2, 3, 5]', "Error in list representation"
    
    # Redirect stdout to capture print output for comparison
    import io
    import sys
    capturedOutput = io.StringIO()                  
    sys.stdout = capturedOutput                     
    my_list.print_sorted()                          
    sys.stdout = sys.__stdout__                     
    assert capturedOutput.getvalue().strip() == '[1, 2, 3, 4, 5]', "Sorted output does not match"

if __name__ == "__main__":
    test_my_list()
    print("All tests passed.")
